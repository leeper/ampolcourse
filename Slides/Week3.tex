\frame{\tableofcontents[currentsection]}

\frame{
\frametitle{Congress as Institution(s)}
	\begin{itemize}\itemsep2em
		\item Constitution was a minimal framework
		\item Distinct patterns of path-dependent institutionalization in each chamber
		\item Institutions structure strategic interactions
	\end{itemize}
	
}

\frame{
\frametitle{The legislative process}
	\begin{itemize}\itemsep2em
		\item \url{https://www.youtube.com/watch?v=H-eYBZFEzf8}
	\end{itemize}
	
}

\frame{
\frametitle{The legislative process}
	\begin{itemize}\itemsep2em
		\item Each chamber can initiate legislation
		\item Chamber delegates to committees
		\item Committees revise and report to floor
		\item Chambers must agree on legislation for it to be adopted
		\item Conflicts resolved via ``conference committee''
	\end{itemize}
	
}

\frame{
\frametitle{Strategic models of Congressional activity}
	\begin{itemize}\itemsep2em
		\item Decision making operates around pivots:
		    \begin{itemize}
		    \item Chamber medians
		    \item Committee medians
		    \item Presidential preference
		    \item Status quo
		    \end{itemize}
		\item Legislation is proposed and voted on strategically
		\item Big question: How much and in what ways do parties (versus members' ideology) matter?
	\end{itemize}
	
}

\frame{
\frametitle{Common features of both chambers}
	\begin{itemize}\itemsep2em
		\item<1-> Partisan leadership
		\item<2-> Seniority
		\item<3-> Widespread majority rule
		\item<4-> Floor debate and committee debate
		\item<5-> Polarization
		\item<6-> Staff and professionalization
	\end{itemize}
	
}

\frame{
\frametitle{House of Representatives}
	\begin{itemize}\itemsep2em
		\item<1-> Larger chamber (435 members)
		\item<2-> Individual members less powerful
		\item<3-> Constrained floor debate
		\item<4-> Power is held by leadership (esp. Speaker)
		\item<5-> Rules Committee
	\end{itemize}
	
}

\frame{
\frametitle{Senate}
	\begin{itemize}\itemsep2em
		\item<1-> Smaller chamber (100 members)
		\item<2-> Power is diffuse
		\item<3-> Unconstrained floor debate
		\item<4-> Unlimited amendments and no germaneness rules
		\item<5-> Minority powers
		    \begin{itemize}
		    \item<5-> Filibuster
		    \item<5-> Holds
		    \end{itemize}
	\end{itemize}
	
}

\frame{
\frametitle{Discussion}
	\begin{itemize}\itemsep2em
		\item<1-> How does Congress compare to other legislative systems?
		\item<2-> What consequences do these institutions have for American politics compared to other political systems?
		\item<3-> How well do these institutions function?
	\end{itemize}
	
}
