\section{States and Federalism}
\frame{\tableofcontents[currentsection]}

\frame{
	\frametitle{Federalism}
	\begin{itemize}\itemsep1em
		\item National government
		\item States
		\item Counties
		\item Cities
		\item Regional (within-state) governments
		\item Direct democracy
	\end{itemize}	
}


\frame{
	\frametitle{Between-State Institutional Variations}
    All states have:
	\begin{itemize}\itemsep1em
		\item Governor as executive
		\item Bicameral legislature (except NE)
		\item Judiciary (but form varies)
	\end{itemize}	
}

\frame{
	\frametitle{Between-State Institutional Variations}
    Lots of variation on:
	\begin{itemize}\itemsep1em
		\item Separation of authority between state, counties, and cities
		\item Role of counties
		\item Presence and types of regional governments
		\item City-level authority and jurisdiction (related to size)
		\item How schools are governed
	\end{itemize}	
}

\frame{
	\frametitle{Between-State Policy Variations}
	\begin{itemize}\itemsep1em
		\item<2-> Congressional laws apply universally across states
		\item<2-> States have jurisdiction in all other matters
		\item<2-> Some overlapping jurisdiction between state and national gov'ts
		\item<3-> Federal authority through ``block grants''
	\end{itemize}	
}

\frame{
	\frametitle{Example: Taxes}
	\begin{itemize}\itemsep1em
		\item National government can only raise funds through income tax and tariffs
		\item States can tax almost anything
		\item Local governments have limited, but variable, taxing powers
	\end{itemize}	
}

\section{Direct Democracy}
\frame{\tableofcontents[currentsection]}

\frame{
	\frametitle{Background}
	\begin{itemize}\itemsep1em
		\item No national direct democracy
		\item Direct election of Senators (17th Amend.; 1913)
		\item Most prevalent in Western states
    		\begin{itemize}
        		\item esp. CA; also AZ, CO, MT, NV, ND, OR
    		\end{itemize}
    	\item All states (except DE) have referenda on state constitutional amendments
	\end{itemize}	
}

\frame{
	\frametitle{Forms of Direct Democracy}
	\begin{itemize}\itemsep1em
		\item Legislature-initiated amendments (49 states)
		\item Legislature-initiated statutes (24)
		\item Citizen-initiated constitutional amendments (18)
		\item Citizen-initiated statutes (21)
    		\begin{itemize}
        		\item Direct (referendum is law-making)
        		\item Indirect (referendum is referring only) 
    		\end{itemize}
		\item ``Veto referendum'' (25)
		\item Recall of elected officials (18)
	\end{itemize}	
}
